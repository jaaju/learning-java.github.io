 1   // Finds the lowest recorded temperature from file at dataFilePath.                                                              
 2   // Since this looks for the "minimum" and the daily low <= daily high,                                                           
 3   // only the daily low field is used in the algorithm.                                                                            
 4   private static int findLowest(String dataFilePath)                                                                               
 5           throws Exception
 6   {
 7     BufferedReader dataFile
 8             = new BufferedReader(new FileReader(dataFilePath));
 9 
10     int m = Integer.MAX_VALUE;                                                                                                     
11     String line = dataFile.readLine();
12     while (line != null) {
13       int firstCommaPos = line.indexOf(',');
14       int nextCommaPos = line.indexOf(',', firstCommaPos + 1);
15       String sTemp = line.substring(firstCommaPos + 1, nextCommaPos);                                                              
16       int temperatureRecorded = Integer.parseInt(sTemp);
17       if (temperatureRecorded < m) {                                                                                               
18         m = temperatureRecorded;
19       }
20       line = dataFile.readLine();
21     }
22     return m;
23   }
